## 一、技術選型

| 層次 | 選擇 | 理由 |
|------|------|------|
| 框架 | Next.js 15 App Router | SSR + 快速路由切換，`layout.tsx` 實現巢狀 UI |
| UI 元件 | shadcn/ui + Tailwind CSS | 符合 `01-專案目標.md` 規範；無頭元件易於客製 |
| 伺服器狀態 | TanStack Query v5 | API 快取、背景輪詢（索引狀態、逐字稿）、樂觀更新 |
| 表單驗證 | React Hook Form + Zod | 型別安全，與後端 schema 共用驗證規則 |
| 圖示 | Lucide React | shadcn/ui 預設使用，風格一致 |
| 檔案上傳 | react-dropzone | 拖拉、點擊、剪貼簿上傳 |
| 通知 | sonner | shadcn/ui 整合的 Toast |

### 認證 Token 策略（與 04-API設計.md 補充）

```
useSession() → session.vexaToken（Vexa API token）
  ↓
呼叫 App Backend 時：Authorization: Bearer <vexaToken>
  ↓
Backend 驗證：查詢 public.api_tokens WHERE token = vexaToken → 取得 user_id
```

這比 NextAuth JWT 更簡單：`vexaToken` 已在 `session` 中，無需額外處理。

## 四、自訂 Hook 設計

```typescript
// ── 專案 ────────────────────────────────────────────

// 取得使用者的所有專案清單
function useProjects(): UseQueryResult<ProjectListResponse>

// 取得單一專案詳細資訊（含 permissions）
function useProject(projectId: string): UseQueryResult<ProjectDetail>

// 建立專案
function useCreateProject(): UseMutationResult

// 刪除專案
function useDeleteProject(): UseMutationResult

// ── 資料 ────────────────────────────────────────────

// 取得資料清單（第一次載入 + 手動重新整理）
function useMaterials(projectId: string): UseQueryResult<PaginatedMaterials>

// 上傳檔案（含 multipart/form-data）
function useUploadMaterial(projectId: string): UseMutationResult

// 刪除檔案
function useDeleteMaterial(projectId: string): UseMutationResult

// 單一材料的索引狀態輪詢（直到 COMPLETED/FAILED 才停止）
function useMaterialStatus(projectId: string, materialId: string) {
  return useQuery({
    queryKey: ['material', materialId],
    queryFn: () => api.getMaterial(projectId, materialId),
    refetchInterval: (query) => {
      const status = query.state.data?.indexingStatus
      return status === 'PROCESSING' || status === 'PENDING' ? 5000 : false
    }
  })
}

// ── 會議 ────────────────────────────────────────────

// 取得會議清單
function useMeetings(projectId: string): UseQueryResult<PaginatedMeetings>

// 取得單一會議（含摘要生成輪詢）
function useMeeting(projectId: string, meetingId: string) {
  return useQuery({
    queryKey: ['meeting', meetingId],
    queryFn: () => api.getMeeting(projectId, meetingId),
    refetchInterval: (query) => {
      const m = query.state.data
      // Bot 加入中：繼續輪詢
      if (m?.status === 'PENDING') return 3000
      // 會議進行中：繼續輪詢以偵測 ACTIVE → ENDED 狀態轉換
      // （Bot 自動離會後才知道 status 變 ENDED，父元件依此切換 isActive → false，停止逐字稿輪詢）
      if (m?.status === 'ACTIVE') return 5000
      // 摘要生成中：繼續輪詢直到 summary 填入
      if (m?.status === 'ENDED' && !m?.summary) return 5000
      // ENDED（已有摘要）或 FAILED：終止態，停止輪詢
      // FAILED 由 MeetingFailedBanner 處理，顯示靜態錯誤提示，無重試按鈕
      return false
    }
  })
}

// 進行中會議的即時逐字稿輪詢
// ⚠️ ACTIVE 會議的 API 使用 since_start_time（float）游標，非 since_id（integer）
// 04-API設計.md：GET .../transcriptions?since_start_time=X 回傳 start_time > X 的 segments
function useLiveTranscriptions(
  projectId: string,
  meetingId: string,
  isActive: boolean   // 只有 ACTIVE 才輪詢
) {
  const [sinceStartTime, setSinceStartTime] = useState<number | null>(null)
  const [allSegments, setAllSegments] = useState<TranscriptSegment[]>([])

  useQuery({
    queryKey: ['transcriptions-live', meetingId],
    queryFn: async () => {
      const res = await api.getTranscriptions(projectId, meetingId, { sinceStartTime })
      if (res.items.length > 0) {
        setAllSegments(prev => [...prev, ...res.items])
        setSinceStartTime(res.items.at(-1)!.startTime)  // float（Vexa 的 start_time）
      }
      return res
    },
    enabled: isActive,
    refetchInterval: 3000,
  })

  return allSegments
}

// ── 權限 ────────────────────────────────────────────

// 從 project 資料推導當前使用者的操作權限
function usePermissions(projectId: string): UserPermissions {
  const { data: project } = useProject(projectId)
  return project?.permissions ?? {
    canView: false, canEdit: false,
    canDelete: false, canManage: false, canMeeting: false
  }
}
```

## 五、API Client 實作

```typescript
// lib/api-client.ts
import { getSession } from 'next-auth/react'

const BASE_URL = process.env.NEXT_PUBLIC_API_URL ?? 'http://localhost:4000'

async function getAuthHeader(): Promise<HeadersInit> {
  const session = await getSession()
  // session.vexaToken 由 NextAuth session callback 注入（Vexa 原本即有）
  const token = (session as any)?.vexaToken
  if (!token) throw new Error('Not authenticated')
  return { Authorization: `Bearer ${token}` }
}

export const apiClient = {
  async get<T>(path: string): Promise<T> {
    const headers = await getAuthHeader()
    const res = await fetch(`${BASE_URL}${path}`, { headers })
    if (!res.ok) throw await res.json()
    return res.json()
  },

  async post<T>(path: string, body?: unknown): Promise<T> {
    const headers = await getAuthHeader()
    const res = await fetch(`${BASE_URL}${path}`, {
      method: 'POST',
      headers: { ...headers, 'Content-Type': 'application/json' },
      body: body ? JSON.stringify(body) : undefined,
    })
    if (!res.ok) throw await res.json()
    return res.json()
  },

  async postForm<T>(path: string, formData: FormData): Promise<T> {
    const headers = await getAuthHeader()  // 注意：multipart 不設 Content-Type，讓瀏覽器自動設
    const res = await fetch(`${BASE_URL}${path}`, {
      method: 'POST',
      headers,  // 不包含 Content-Type
      body: formData,
    })
    if (!res.ok) throw await res.json()
    return res.json()
  },

  async patch<T>(path: string, body: unknown): Promise<T> {
    const headers = await getAuthHeader()
    const res = await fetch(`${BASE_URL}${path}`, {
      method: 'PATCH',
      headers: { ...headers, 'Content-Type': 'application/json' },
      body: JSON.stringify(body),
    })
    if (!res.ok) throw await res.json()
    return res.json()
  },

  async delete(path: string): Promise<void> {
    const headers = await getAuthHeader()
    const res = await fetch(`${BASE_URL}${path}`, { method: 'DELETE', headers })
    if (!res.ok) throw await res.json()
  },
}
```

## 七、PermissionGuard 元件

用於條件式渲染需要特定權限才能看到的 UI：

```typescript
// components/permission-guard.tsx
'use client'
import { usePermissions } from '@/hooks/use-permissions'
import type { UserPermissions } from '@/types/api'

interface Props {
  projectId: string
  require: keyof UserPermissions
  children: React.ReactNode
  fallback?: React.ReactNode
}

export function PermissionGuard({ projectId, require, children, fallback = null }: Props) {
  const permissions = usePermissions(projectId)
  if (!permissions[require]) return <>{fallback}</>
  return <>{children}</>
}
```

```tsx
</PermissionGuard>
```
