## 三、程式進入點（`src/index.ts`）

```typescript
import { Hono } from 'hono'
import { cors } from 'hono/cors'
import { authMiddleware } from './middleware/auth'
import { errorHandler } from './middleware/error-handler'
import { requestLogger } from './middleware/logger'
import { registerRoutes } from './routes'
import { startIndexingPoller } from './jobs/indexing-poller'
import { restoreActiveSessions } from './sessions/session-manager'

const app = new Hono()

// ── 全域 Middleware ──────────────────────────────
app.use('*', requestLogger())
app.use('*', cors({ origin: process.env.APP_CORS_ORIGINS!.split(',') }))
app.use('*', authMiddleware)   // 所有路由都需要認證（路由無 /api/ 前綴，不可用 /api/*）

// ── 路由 ─────────────────────────────────────────
registerRoutes(app)

// ── 錯誤處理 ─────────────────────────────────────
app.onError(errorHandler)

// ── Background Jobs ──────────────────────────────
startIndexingPoller()         // 每 30 秒輪詢 Dify 索引狀態

// ── 啟動時恢復進行中的 Session（服務重啟後） ──────
await restoreActiveSessions() // 查詢 DB 中 status=ACTIVE 的會議，重建 WebSocket

export default { port: Number(process.env.APP_PORT ?? 4000), fetch: app.fetch }
```

## 四、認證 Middleware（`src/middleware/auth.ts`）

```typescript
import type { Context, MiddlewareHandler } from 'hono'
import { prisma } from '../lib/prisma'

export const authMiddleware: MiddlewareHandler = async (c, next) => {
  const token = c.req.header('Authorization')?.replace('Bearer ', '')
  if (!token) {
    return c.json({ error_code: 'UNAUTHORIZED', message: '缺少 Authorization header' }, 401)
  }

  const apiToken = await prisma.$queryRaw<Array<{ user_id: number; id: number; scopes: string[] }>>`
    SELECT user_id, id, scopes FROM public.api_tokens
    WHERE token = ${token}
      AND (expires_at IS NULL OR expires_at > NOW())
    LIMIT 1
  `
  // scopes 為 PostgreSQL text[]，Prisma $queryRaw 回傳 JavaScript string[]
  if (!apiToken.length) {
    return c.json({ error_code: 'UNAUTHORIZED', message: '無效的 token' }, 401)
  }

  const user = await prisma.$queryRaw<Array<{
    id: number; email: string; name: string | null; max_concurrent_bots: number
  }>>`
    SELECT id, email, name, max_concurrent_bots FROM public.users WHERE id = ${apiToken[0].user_id} LIMIT 1
  `

  // 注入到 context，後續 handler 直接使用
  c.set('vexaApiTokenId', apiToken[0].id)          // POST /meetings 存 creatorApiTokenId 所需
  c.set('vexaTokenScopes', apiToken[0].scopes)     // 供 handler scope 前置檢查（見下方輔助函式）
  c.set('vexaUserId', user[0].id)
  c.set('userEmail', user[0].email)
  c.set('userName', user[0].name)
  c.set('maxConcurrentBots', user[0].max_concurrent_bots)
  c.set('vexaToken', token)

  await next()
}
```

> **設計決策**：使用 `prisma.$queryRaw` 而非 Prisma model 操作，是因為 `public` schema 的表為
> Vexa 管理的唯讀引用，不建立對應的 Prisma model（避免 migrate 衝突）。
> 實際部署時可視需要改用 Prisma 的 `prisma db pull` 結果。

**scope 前置檢查輔助函式**（供需要呼叫 Vexa API 的 handler 使用）：

```typescript
// src/middleware/auth.ts（輔助函式，與 authMiddleware 同文件）

const BOT_REQUIRED_SCOPES = ['bot', 'browser', 'tx'] as const

/**
 * 在呼叫 Vexa Bot API 前驗證 token scope 是否足夠。
 * 在 createMeeting handler 的步驟①呼叫，在 DB 寫入前快速失敗。
 *
 * ⚠️ 為何需要明確檢查：Vexa api-gateway 的 ROUTE_SCOPES 強制要求：
 *   /bots → {"bot","browser"}（POST /bots 邀請 Bot）
 *   /transcripts → {"tx"}（GET /transcripts 取逐字稿）
 * 若缺少 scope，Vexa 回傳 403 且 meetbot 難以從錯誤訊息判別根因；
 * 前置檢查確保錯誤在 meetbot 層給出明確的 INSUFFICIENT_SCOPE 訊息。
 */
export function requireBotScopes(c: Context): Response | null {
  const scopes: string[] = c.get('vexaTokenScopes') ?? []
  const missing = BOT_REQUIRED_SCOPES.filter(s => !scopes.includes(s))
  if (missing.length === 0) return null
  return c.json({
    error_code: 'INSUFFICIENT_SCOPE',
    message: `此 token 缺少邀請 Bot 所需的 scope（需要 bot、browser、tx）`,
    details: { required: BOT_REQUIRED_SCOPES, missing },
  }, 403)
}
```

### 5.2 Session Manager（`session-manager.ts`）

每個活躍的 `MeetingSession` 使用**邀請者的 `creatorVexaToken`** 建立專屬的 WebSocket 連線。

> **設計決策（修正 Bug #1）**：Vexa `/ws/authorize-subscribe` 的授權邏輯為
> `Meeting.user_id == current_user.id`（見 `collector/endpoints.py` lines 355–402）。
> 若使用單一「服務級」token，只能訂閱**該 token 所屬使用者**所建立的會議，
> 無法接收其他使用者建立的會議事件，導致跨使用者場景完全失效。
>
> 修正方案：每個 Session 用**邀請者自己的 token** 建立獨立 WebSocket，
> 代價是 N 個進行中會議 = N 條 WS 連線（可接受，因 `max_concurrent_bots` 預設 = 1）。

```typescript
// Vexa /ws 的三條 Redis channel（訂閱後自動接收）
// tc:meeting:{id}:mutable   → 逐字稿 segments（語音輸入）
// bm:meeting:{id}:status    → Bot 狀態變化（active / completed / failed 等）
// va:meeting:{id}:chat      → 聊天室訊息（來自 Meet 的文字輸入）

/**
 * 建立 WS Session 並訂閱 Vexa 三條 channel。
 * ⚠️ 若拋出例外（WS 連線失敗），呼叫方（meeting.service.ts）需：
 *   1. 呼叫 Vexa DELETE /bots/{platform}/{nativeMeetingId} 撤銷已建立的 Bot
 *   2. 將 MeetingInstance 保留 PENDING 狀態（不更新 DB 至 ACTIVE）
 *
 * 本函式成功返回後，呼叫方**只更新 vexaMeetingId / vexaNativeMeetingId / creatorApiTokenId**，
 * DB 維持 PENDING。DB 轉 ACTIVE 由 handleBotStatusChange 收到 Vexa WS 的
 * {type:"meeting.status", payload:{status:"active"}} 事件後自動執行，呼叫方不需主動設定。
 * 原因：Vexa 的 _find_active_meeting 要求 meeting.status == "active"，
 *       若提前設 ACTIVE，在 Bot 尚未入會期間的 /speak、/chat 呼叫將回傳 404。
 */
export async function createSession(meetingInstanceId: string, params: {
  vexaMeetingId: number
  platform: string              // 固定 "google_meet"
  nativeMeetingId: string       // 例如 "abc-defg-hij"
  difyDatasetId: string | null  // null = 無關聯專案，Q&A 功能停用
  creatorVexaToken: string
}): Promise<void> {
  // 1. 用邀請者的 token 建立此 session 專屬的 WebSocket 連線
  const ws = new WebSocket(`${process.env.VEXA_WS_URL}/ws`, {
    headers: { 'X-API-Key': params.creatorVexaToken },
  })

  // 2. 初始化 session 物件（先存入 Map，讓 message handler 可透過 closure 取得）
  // difyDatasetId 可為 null（無關聯專案時），此時 Q&A 功能停用
  const session: MeetingSession = {
    meetingInstanceId,
    vexaMeetingId: params.vexaMeetingId,
    platform: params.platform,
    nativeMeetingId: params.nativeMeetingId,
    difyDatasetId: params.difyDatasetId,  // null = 無關聯專案
    creatorVexaToken: params.creatorVexaToken,
    isSpeaking: false,
    lastWakeAt: 0,
    processedSegmentIds: new Set(),    // 防止喚醒詞重複觸發（上限 MAX_PROCESSED_SEGMENT_IDS）
    wsConnection: ws,
    difyConversationId: null,          // 多輪對話 ID，由 resolveAnswer 在首次問答後填入
    lastQuestionAt: 0,                 // 初始為 0，確保第一次問問題不會誤判為剛剛有對話
  }
  activeSessions.set(meetingInstanceId, session)

  // 3. 等待連線建立後，訂閱此會議的三條 channel
  await new Promise<void>((resolve, reject) => {
    ws.once('open', () => {
      ws.send(JSON.stringify({
        action: 'subscribe',
        meetings: [{ platform: params.platform, native_id: params.nativeMeetingId }],
      }))
      resolve()
    })
    ws.once('error', (err) => reject(err))
  })

  // 4. 掛載訊息處理器（透過 closure 直接取得 session，不需 Map 遍歷查詢）
  ws.on('message', (data) => {
    const msg = JSON.parse(data.toString())
    const s = activeSessions.get(meetingInstanceId)  // closure：meetingInstanceId 已固定
    if (!s) return
    handleVexaWsMessage(s, msg)
  })

  // 5. 非主動關閉時自動重連（主動呼叫 closeSession 會先從 Map 移除，阻止重連）
  ws.on('close', () => {
    logger.warn({ meetingInstanceId }, 'Per-session WS disconnected')
    if (activeSessions.has(meetingInstanceId)) {
      setTimeout(() => createSession(meetingInstanceId, params), 3000)
    }
  })
  ws.on('error', (err) => logger.error({ err, meetingInstanceId }, 'Per-session WS error'))
}

export async function closeSession(meetingInstanceId: string): Promise<void> {
  const session = activeSessions.get(meetingInstanceId)
  if (!session) return

  // 先從 Map 移除，確保 ws.on('close') 不觸發重連
  activeSessions.delete(meetingInstanceId)

  session.wsConnection.send(JSON.stringify({
    action: 'unsubscribe',
    meetings: [{ platform: session.platform, native_id: session.nativeMeetingId }],
  }))
  session.wsConnection.close()
}

// ── Vexa /ws 訊息路由（修正 Bug #3：使用正確的 Vexa 訊息型別字串）──

function handleVexaWsMessage(session: MeetingSession, msg: any): void {
  switch (msg.type) {
    case 'transcript.bundle': {
      // 新格式（tc:meeting:{id}:mutable）：由 process_transcript_bundle 發布
      // ⚠️ 只處理 confirmed[]，pending[] 為語音識別中間態，不觸發喚醒詞
      const confirmedSegs: any[] = msg.confirmed ?? []
      for (const seg of confirmedSegs) {
        if (!seg.text?.trim() || !seg.segment_id) continue
        handleTranscriptSegment(session, {
          segment_id: seg.segment_id,
          text:       seg.text,
          speaker:    seg.speaker || msg.speaker || '',
          start_time: seg.start  || seg.start_time  || 0,
          end_time:   seg.end    || seg.end_time    || 0,
        })
      }
      break
    }
    case 'transcript.segments': {
      // 舊格式（legacy）：由 process_stream_message 發布
      // ⚠️ 只處理 completed: true 的 segment（過濾進行中的識別片段）
      const segs: any[] = msg.segments ?? []
      for (const seg of segs) {
        if (!seg.completed || !seg.text?.trim() || !seg.segment_id) continue
        handleTranscriptSegment(session, seg)
      }
      break
    }
    case 'meeting.status': {
      // bm:meeting:{id}:status → Bot 狀態變化
      // ⚠️ 注意：Vexa 實際型別為 "meeting.status"，非 "meeting_status"
      const status = msg.payload?.status
      if (status) handleBotStatusChange(session, status)
      break
    }
    case 'chat.new_message': {
      // va:meeting:{id}:chat → 聊天室訊息（跳過 Bot 自發訊息，避免無限迴圈）
      // ⚠️ Vexa 實際格式（vexa-bot/core/src/services/chat.ts:186–199）：
      //   { type: "chat.new_message", payload: { sender, text, timestamp, is_from_bot }, ts, meeting }
      //   is_from_bot 在 payload 子物件內，不在根層級
      if (!msg.payload?.is_from_bot) handleChatMessage(session, msg.payload)
      break
    }
    // 'subscribed', 'pong', 'error' 等控制訊息忽略
  }
}

// ── Bot 自動離會偵測 ─────────────────────────────────────────────

async function handleBotStatusChange(session: MeetingSession, status: string): Promise<void> {
  // Vexa meeting.status 狀態機：
  //   requested → joining → awaiting_admission → active（Bot 成功入會）
  //                                             → completed（正常結束）/ failed（異常）

  if (status === 'active') {
    // Bot 成功進入會議 → 將 meetbot DB 從 PENDING 轉為 ACTIVE
    // ⚠️ 此處才設 ACTIVE（而非 createSession() 完成時），確保 Vexa meeting.status == "active"
    //    後 /speak、/chat 才可用（Vexa _find_active_meeting 以此欄位作為可用性守門）
    logger.info({ meetingInstanceId: session.meetingInstanceId }, 'Bot entered meeting → marking ACTIVE')
    await prisma.meetingInstance.update({
      where: { id: session.meetingInstanceId },
      data: { status: 'ACTIVE', startedAt: new Date() },
    })

    // 歡迎訊息在此發送：Vexa meeting.status 確認為 "active" 後 /chat 才可用
    // ⚠️ 不可在 createSession() 後立即送——Bot 仍在 joining/awaiting_admission 階段，
    //    Vexa _find_active_meeting 回 404（voice_agent.py:47）
    const welcomeMsg = session.difyDatasetId
      ? '嗨大家好！我是蜜塔（Meeta），你們今天的會議小幫手 🎉\n\n你可以用語音或聊天室呼叫我：\n  語音：說「蜜塔」或「小幫手」，再接上你的問題\n  聊天室：輸入「蜜塔」或「小幫手」，再接上你的問題\n\n我會根據此專案上傳的資料來回答問題，例如：\n「蜜塔，請問去年 Q3 目標是什麼？」\n\n有問題就找我！'
      : '嗨大家好！我是蜜塔（Meeta），你們今天的會議小幫手 🎉\n\n你可以用語音或聊天室呼叫我：\n  語音：說「蜜塔」或「小幫手」，再接上你的問題\n  聊天室：輸入「蜜塔」或「小幫手」，再接上你的問題\n\n我會根據本次會議的逐字稿記錄來回答問題，例如：\n「蜜塔，剛才提到的時程安排是什麼？」\n\n有問題就找我！'
    await vexaClient.chatSend(
      session.platform,
      session.nativeMeetingId,
      session.creatorVexaToken,
      { text: welcomeMsg },
    ).catch(err => logger.error({ err, meetingInstanceId: session.meetingInstanceId }, 'welcome chat failed'))

    return
  }

  // Vexa 的 terminal states：completed（正常結束）、failed（異常）
  // 觸發條件包含：使用者踢出（evicted）、無人（left_alone）、逾時（awaiting_admission_timeout）
  // 以及我方主動呼叫 DELETE /bots（stopping → completed）
  const isTerminal = ['completed', 'failed'].includes(status)
  if (!isTerminal) return

  logger.info({ meetingInstanceId: session.meetingInstanceId, vexaStatus: status }, 'Bot left meeting')

  // 傳入 vexaStatus，讓 handleSessionClose 決定 meetbot DB 最終狀態：
  //   completed → ENDED（正常結束，有逐字稿可摘要）
  //   failed    → FAILED（異常終止，前端顯示「建立失敗」而非「已結束」）
  await handleSessionClose(session.meetingInstanceId, status)
}

async function handleSessionClose(meetingInstanceId: string, vexaStatus?: string): Promise<void> {
  const session = activeSessions.get(meetingInstanceId)
  if (!session) return

  // 依 Vexa 回報的結果決定 meetbot 的終態：
  //   vexaStatus === 'failed' → FAILED（異常：Bot 在 active 前即被踢出或逾時）
  //   其他（completed 或 undefined：使用者主動呼叫 POST bot/leave）→ ENDED
  const meetbotFinalStatus = vexaStatus === 'failed' ? 'FAILED' : 'ENDED'

  await prisma.meetingInstance.update({
    where: { id: meetingInstanceId },
    data: { status: meetbotFinalStatus, endedAt: new Date() },
  })

  // 僅在正常結束（ENDED）時觸發摘要工作流；FAILED 代表 Bot 未曾 active，逐字稿可能為空
  // ⚠️ 在 closeSession() 之前捕捉 session 資料：closeSession 會立刻從 Map 刪除 session，
  //    generateSummaryAsync 是 fire-and-forget，若在其內部才取 session 資料會取到 undefined。
  if (meetbotFinalStatus === 'ENDED') {
    generateSummaryAsync({
      meetingInstanceId,
      platform: session.platform,
      nativeMeetingId: session.nativeMeetingId,
      creatorVexaToken: session.creatorVexaToken,
      difyDatasetId: session.difyDatasetId,
    })
  }

  // 清理 session
  await closeSession(meetingInstanceId)
}

// ── 摘要生成（非同步，fire-and-forget）──────────────────────────────
//
// 問題根因：Vexa 的 meeting.status:completed 事件由 bot 容器觸發（bot 離開會議），
// 但 ASR pipeline（Whisper worker）為獨立非同步程序，可能仍有佇列音訊尚未轉文字。
// 若直接取逐字稿，最後幾秒音訊會被截斷。
//
// 修正策略：
//   1. 初始等待（5 秒）：給 Whisper worker 基本的 flush 時間
//   2. 輪詢穩定（每 3 秒比對 segment count）：連續 2 次相同視為穩定
//   3. 硬超時（30 秒）：無論是否穩定，超時後用現有逐字稿直接生成

const SUMMARY_INITIAL_WAIT_MS  = 5_000   // ASR pipeline flush 初始緩衝
const SUMMARY_POLL_INTERVAL_MS = 3_000   // 每 3 秒確認一次 segment count
const SUMMARY_STABLE_POLLS     = 2       // 連續 N 次 count 相同視為穩定
const SUMMARY_TIMEOUT_MS       = 30_000  // 超時後用現有逐字稿直接生成

async function waitForTranscriptStable(
  platform: string,
  nativeMeetingId: string,
  token: string,
): Promise<VexaTranscriptSegment[]> {
  await new Promise(r => setTimeout(r, SUMMARY_INITIAL_WAIT_MS))

  const deadline = Date.now() + SUMMARY_TIMEOUT_MS
  let prevCount   = -1
  let stableCount = 0
  let lastSegments: VexaTranscriptSegment[] = []

  while (Date.now() < deadline) {
    lastSegments = await vexaClient.getTranscriptions(platform, nativeMeetingId, token)
    const count  = lastSegments.length

    if (count === prevCount) {
      if (++stableCount >= SUMMARY_STABLE_POLLS) return lastSegments   // 穩定，可繼續
    } else {
      stableCount = 0
      prevCount   = count
    }

    await new Promise(r => setTimeout(r, SUMMARY_POLL_INTERVAL_MS))
  }

  // 超時（30 秒後仍未穩定）：用目前最後取得的逐字稿直接生成
  logger.warn(
    { nativeMeetingId, segmentCount: lastSegments.length },
    'transcript stabilization timeout, proceeding with available segments',
  )
  return lastSegments
}

async function generateSummaryAsync(params: {
  meetingInstanceId: string
  platform: string
  nativeMeetingId: string
  creatorVexaToken: string
  difyDatasetId: string | null
}): Promise<void> {
  try {
    // 等待 ASR 穩定後再取逐字稿（核心修正點）
    const segments = await waitForTranscriptStable(
      params.platform,
      params.nativeMeetingId,
      params.creatorVexaToken,
    )

    if (!segments.length) {
      logger.info({ meetingInstanceId: params.meetingInstanceId }, 'no transcript, skipping summary')
      return
    }

    const meeting = await prisma.meetingInstance.findUnique({
      where: { id: params.meetingInstanceId },
      select: { title: true },
    })

    const transcript = segments
      .map(seg => `[${seg.speaker || '參與者'}]: ${seg.text}`)
      .join('\n')

    const { summary, actionItems } = await dify.generateSummary({
      transcript,
      meetingName: meeting?.title ?? '本次會議',
      meetingInstanceId: params.meetingInstanceId,
    })

    await prisma.meetingInstance.update({
      where: { id: params.meetingInstanceId },
      data: { summaryText: summary, actionItems },
    })

    logger.info(
      { meetingInstanceId: params.meetingInstanceId, actionItemCount: actionItems.length },
      'meeting summary generated',
    )
  } catch (err) {
    logger.error({ err, meetingInstanceId: params.meetingInstanceId }, 'generateSummaryAsync failed')
  }
}
```

### 5.3 喚醒詞偵測（`wake-word-detector.ts`）

語音輸入和聊天室輸入來自 Vexa WS 的**不同 channel**，需分開處理：

```typescript
// 中文：蜜塔/密塔/祕塔/秘塔/迷塔、小幫手
// 英文：Meeta/meeta/Mita/mita（語音辨識可能輸出英文拼寫）
const WAKE_WORD_REGEX = /[蜜密祕秘迷][塔搭]|小幫手|[Mm]eeta|[Mm]ita/
const DEBOUNCE_MS = 2000          // 2 秒內不重複觸發（跨語音和聊天室共用）
const MAX_PROCESSED_SEGMENT_IDS = 5000  // 防記憶體無界成長（長會議約 8 小時安全邊際）

// ── 語音輸入（來自 tc:meeting:{id}:mutable channel）────────────

export async function handleTranscriptSegment(
  session: MeetingSession,
  segment: VexaTranscriptSegment   // 欄位：text, segment_id, speaker, start_time, end_time 等
): Promise<void> {
  // 防重複：同一 segment_id 已處理過（Vexa 可能重送 mutable segments）
  if (!segment.segment_id || session.processedSegmentIds.has(segment.segment_id)) return

  // 防記憶體無界成長：超過上限時保留最新的一半（接受極小概率的重複觸發風險）
  if (session.processedSegmentIds.size >= MAX_PROCESSED_SEGMENT_IDS) {
    const ids = [...session.processedSegmentIds]
    session.processedSegmentIds = new Set(ids.slice(Math.floor(ids.length / 2)))
  }
  session.processedSegmentIds.add(segment.segment_id)

  const match = WAKE_WORD_REGEX.exec(segment.text)
  if (!match) return

  // 防重複觸發：2 秒內只觸發一次
  const now = Date.now()
  if (now - session.lastWakeAt < DEBOUNCE_MS) return
  session.lastWakeAt = now

  const question = segment.text.slice(match.index + match[0].length).trim()
  if (!question) return

  // 語音輸入 → 語音回覆
  await dispatchQuestion(session, question, 'voice')
}

// ── 聊天室輸入（來自 va:meeting:{id}:chat channel）──────────────

export async function handleChatMessage(
  session: MeetingSession,
  chatMsg: VexaChatMessage   // 欄位：text, sender, timestamp, is_from_bot 等
): Promise<void> {
  // 忽略 Bot 自己發出的訊息，避免無限迴圈
  if (chatMsg.is_from_bot) return

  const match = WAKE_WORD_REGEX.exec(chatMsg.text)
  if (!match) return

  // 防重複觸發：與語音共用同一個 lastWakeAt 計時器
  const now = Date.now()
  if (now - session.lastWakeAt < DEBOUNCE_MS) return
  session.lastWakeAt = now

  const question = chatMsg.text.slice(match.index + match[0].length).trim()
  if (!question) return

  // 聊天室輸入 → 聊天室回覆
  await dispatchQuestion(session, question, 'chat')
}

// ── 問題分派（根據來源決定回覆管道）───────────────────────────────

// 超過此時間未提問，重置 Dify 對話（避免 conversation 過期 + 避免過長對話拖慢 LLM）
const CONVERSATION_IDLE_RESET_MS = 5 * 60 * 1000  // 5 分鐘

/**
 * 統一問答入口：選擇查詢路徑並管理 Dify 多輪對話 ID。
 *   有知識庫（difyDatasetId 非 null）→ Dify Chatflow RAG，帶 conversation_id 實現多輪連貫
 *   無知識庫（獨立會議）→ 最近 30 段逐字稿 + Claude 直接回答（無多輪上下文）
 *
 * Dify conversation_id 管理策略：
 *   - 閒置超過 5 分鐘 → 重置 ID，下次問題開新對話
 *   - Dify 回傳錯誤（舊 ID 過期等）→ 自動重置後重試一次（建立新對話）
 */
async function resolveAnswer(session: MeetingSession, question: string): Promise<string> {
  if (!session.difyDatasetId) {
    const { answer } = await answerFromTranscript(session, question)
    return answer
  }

  // 閒置重置：lastQuestionAt = 0 時（初始值）不重置（避免首次問答開新對話後立刻再重置）
  if (session.lastQuestionAt > 0 && Date.now() - session.lastQuestionAt > CONVERSATION_IDLE_RESET_MS) {
    logger.info({ meetingInstanceId: session.meetingInstanceId }, 'Dify conversation reset: idle timeout')
    session.difyConversationId = null
  }

  const callDify = async (conversationId: string | null) =>
    dify.askQuestion({
      datasetId: session.difyDatasetId!,
      question,
      userId: session.meetingInstanceId,
      conversationId,
    })

  try {
    const { answer, conversationId } = await callDify(session.difyConversationId)
    session.difyConversationId = conversationId || session.difyConversationId
    session.lastQuestionAt = Date.now()
    return answer
  } catch (err) {
    // Dify 錯誤（最常見：舊 conversation_id 已過期）→ 重置後重試一次
    if (session.difyConversationId) {
      logger.warn({ meetingInstanceId: session.meetingInstanceId }, 'Dify error, resetting conversation and retrying')
      session.difyConversationId = null
      const { answer, conversationId } = await callDify(null)
      session.difyConversationId = conversationId
      session.lastQuestionAt = Date.now()
      return answer
    }
    throw err
  }
}

async function dispatchQuestion(
  session: MeetingSession,
  question: string,
  source: 'voice' | 'chat'
): Promise<void> {
  const pendingVoice = session.difyDatasetId
    ? '好的，我收到了，正在查詢資料，請稍候。'
    : '好的，我收到了，正在查閱會議記錄，請稍候。'
  const pendingChat = session.difyDatasetId
    ? '收到你的問題，正在查詢資料中……'
    : '收到你的問題，正在查閱會議記錄……'

  // 語音模式：在查詢開始前就鎖住 isSpeaking，防止查詢期間第二次喚醒詞插入
  if (source === 'voice') {
    if (session.isSpeaking) return
    const PROMPT_TEXT = pendingVoice
    const promptEstimatedMs = Math.max(3000, (PROMPT_TEXT.length / 4) * 1000 + 1500)
    session.isSpeaking = true
    const lockTimer = setTimeout(() => { session.isSpeaking = false }, promptEstimatedMs + 10_000)

    try {
      await vexaClient.speak(session.platform, session.nativeMeetingId, session.creatorVexaToken, {
        text: PROMPT_TEXT,
        provider: 'openai',
        voice: 'alloy',
      })

      const answer = await resolveAnswer(session, question)

      clearTimeout(lockTimer)
      const answerEstimatedMs = Math.max(3000, (answer.length / 4) * 1000 + 1500)
      setTimeout(() => { session.isSpeaking = false }, promptEstimatedMs + answerEstimatedMs)

      await vexaClient.speak(session.platform, session.nativeMeetingId, session.creatorVexaToken, {
        text: answer,
        provider: 'openai',
        voice: 'alloy',
      })
    } catch (err) {
      clearTimeout(lockTimer)
      session.isSpeaking = false
      logger.error({ err, meetingInstanceId: session.meetingInstanceId }, 'dispatchQuestion voice failed')
    }
  } else {
    await vexaClient.chatSend(session.platform, session.nativeMeetingId, session.creatorVexaToken, {
      text: pendingChat,
    })

    try {
      const answer = await resolveAnswer(session, question)
      await vexaClient.chatSend(session.platform, session.nativeMeetingId, session.creatorVexaToken, {
        text: answer,
      })
    } catch (err) {
      logger.error({ err, meetingInstanceId: session.meetingInstanceId }, 'dispatchQuestion chat failed')
      await vexaClient.chatSend(session.platform, session.nativeMeetingId, session.creatorVexaToken, {
        text: '抱歉，查詢時發生錯誤，請稍後再試。',
      })
    }
  }
}

// ── 逐字稿 Q&A（無知識庫路徑）──────────────────────────────────────
//
// 以最近 30 段逐字稿為 context，直接送 claude-sonnet-4-6 回答。
// 適用於獨立會議（difyDatasetId === null）。
// 不呼叫 Dify，直接使用 ANTHROPIC_API_KEY 呼叫 Claude API（無多輪上下文）。

async function answerFromTranscript(
  session: MeetingSession,
  question: string,
): Promise<{ answer: string }> {
  const allSegments = await vexaClient.getTranscriptions(
    session.platform,
    session.nativeMeetingId,
    session.creatorVexaToken,
  )
  const recentSegments = allSegments.slice(-30)   // 最近約 5~10 分鐘
  if (!recentSegments.length) {
    return { answer: '目前還沒有足夠的逐字稿內容可以回答，請稍後再試。' }
  }
  const context = recentSegments
    .map(seg => `[${seg.speaker || '參與者'}]: ${seg.text}`)
    .join('\n')

  const message = await anthropic.messages.create({
    model: 'claude-sonnet-4-6',
    max_tokens: 512,
    system: '你是在線的 AI 會議助理蜜塔（Meeta）。請根據提供的會議逐字稿片段，簡潔回答問題。若逐字稿沒有相關資訊，請說明無法從當前會議內容找到答案。',
    messages: [{
      role: 'user',
      content: `以下是近期的會議逐字稿片段：\n\n${context}\n\n請回答：${question}`,
    }],
  })
  const answer = message.content[0].type === 'text' ? message.content[0].text : '抱歉，無法取得回答。'
  return { answer }
}
```

## 五之一、逐字稿服務（`src/services/transcription.service.ts`）

`GET /projects/:projectId/meetings/:meetingId/transcriptions` 的核心邏輯。

### 關鍵約束：必須使用邀請者 token

Vexa `GET /transcripts/{platform}/{nativeId}` 在內部執行：

```python

meeting = Meeting.user_id == current_user.id
```

因此若使用**非邀請者**的 token（例如：有 `canView` 的參與者），Vexa 回傳 404（不是 403）。
必須取出邀請者的 `creatorVexaToken` 來發送請求。

### Token 取得策略

```typescript
/**
 * 取得呼叫 Vexa transcription API 所需的 creatorVexaToken。
 * ACTIVE 時從記憶體取（O(1)）；ENDED/FAILED 時查 DB（Session 已清除）。
 */
async function resolveCreatorToken(
  meetingInstanceId: string,
  creatorApiTokenId: number,
): Promise<string | null> {
  // 1. 優先從記憶體中的活躍 Session 取得（ACTIVE 會議）
  const session = activeSessions.get(meetingInstanceId)
  if (session) return session.creatorVexaToken

  // 2. Session 已消失（ENDED / FAILED）：跨 schema 查詢原始 token 字串
  //    public.api_tokens 在 Vexa 管理的 schema，不可建立 FK constraint，用 raw query 存取
  const rows = await prisma.$queryRaw<Array<{ token: string }>>`
    SELECT token FROM public.api_tokens
    WHERE id = ${creatorApiTokenId}
      AND (expires_at IS NULL OR expires_at > NOW())
    LIMIT 1
  `
  return rows[0]?.token ?? null
}
```

### Handler 骨架

```typescript
export async function getTranscriptions(params: {
  meetingInstanceId: string
  creatorApiTokenId: number
  platform: string
  vexaNativeMeetingId: string
  sinceStartTime?: number
  page?: number
  perPage?: number
}): Promise<TranscriptionResult> {
  const token = await resolveCreatorToken(params.meetingInstanceId, params.creatorApiTokenId)
  if (!token) {
    // token 已過期或遭撤銷——對使用者無法顯示逐字稿，但這不是 404（資源存在）
    throw new AppError('CREATOR_TOKEN_UNAVAILABLE', 503)
  }

  // Vexa REST API 不支援 server-side since_start_time 過濾：取全量後在記憶體裁切
  // ⚠️ O(n) 陷阱：長會議（2 小時 ≈ 2000+ segments）每 3 秒輪詢會傳輸大量重複資料
  //    MVP 可接受；未來可改為 SSE 推送或 backend 快取 cursor
  const all = await vexaClient.getTranscriptions(
    params.platform,
    params.vexaNativeMeetingId,
    token,
  )

  // ⚠️ 使用 >= 而非 >：同一 start 可能有多段（多人同時開口）。
  // >= 避免 cursor 邊界片段被略過；重複片段由前端 segment_id 去重處理。
  const filtered = params.sinceStartTime !== undefined
    ? all.filter(seg => seg.start >= params.sinceStartTime!)
    : all

  // ENDED 會議支援分頁（靜態資料）；ACTIVE 會議直接回全量 filtered
  const page = params.page ?? 1
  const perPage = params.perPage ?? 50
  const items = filtered.slice((page - 1) * perPage, page * perPage)

  return { items, total: filtered.length, page, perPage }
}
```

> **注意**：`creatorApiTokenId`（`MeetingInstance.creatorApiTokenId`）在建立會議時
> 從 `c.get('vexaApiTokenId')`（auth middleware 注入的整數 PK）寫入 DB，確保關聯可追溯。

## 六、Dify 外部服務封裝（`src/lib/dify.ts`）

> **架構決策（v1.3 確認）**：Q&A 透過 **Dify Chatflow**，摘要直接使用 **Claude API**。
>
> ### Q&A：使用 Dify Chatflow（`01-edu2.yml`）
>
> Dify 組員提供的 Chatflow（`docs/01-edu2.yml`）支援在 **runtime 動態傳入 `dataset_id`**，
> 工作流的 HTTP Request node 使用 URL 插值
> `https://api.dify.ai/v1/datasets/{{#1777452815445.dataset_id#}}/retrieve`，
> 完全動態，不需在 Dify 平台 hardcode dataset。
>
> Chatflow 內部流程（優於手動 Retrieve+Claude 的原因）：
> 1. **Gemini 2.5 Flash** 做雙語關鍵字提取（中→中+英，提升混語料召回率）
> 2. **hybrid_search**（語義 + BM25）+ Jina reranker-m0 重排，top_k=8，score ≥ 0.5 過濾
> 3. **Claude Opus 4**（`claude-opus-4-20250514`）依 context 嚴格回答
>
> 因此 `askQuestion` 只需把 `question` 與 `dataset_id` 送給 Chatflow，
> 不需在 meetbot 後端自行組 RAG prompt 或呼叫 Claude API。
>
> ### 摘要：透過 Dify Workflow（`inputs.mode = "summary"`）
>
> 摘要與 Q&A 使用**同一個 Dify App**，以 `inputs.mode` 區分流程：
> - Q&A：`mode = "qa"`，需傳入 `dataset_id`（由 Chatflow 查 Knowledge Base）
> - 摘要：`mode = "summary"`，傳入 `transcript` 與 `meeting_name`（逐字稿已在手，無需 RAG）
>
> ⚠️ **整合前置條件**：`01-edu2.yml` 目前為純 Q&A 流程。
> 需在 Dify App 的 Start 節點加入 `mode`、`transcript`、`meeting_name` 三個輸入變數，
> 並新增 If-Else 條件節點，以 `mode == "summary"` 分支至摘要生成 LLM 節點。
> 摘要 LLM 節點應回傳 JSON：`{"summary": "...", "action_items": [...]}`。
>
> ### 注意：`docs/01-edu2.yml` 已加入 `.gitignore`
>
> 該檔案的 HTTP Request node 原含硬編碼 `Authorization: Bearer <dataset-...>`，
> 已替換為佔位符 `{{DIFY_DATASET_API_KEY}}`，不納入版本控制。
> 匯入 Dify 平台後，請在 HTTP Request node 的 headers 欄位填入實際的 `dataset-...` API Key，
> 或改用 Dify 的環境變數功能（Settings → Environment Variables）注入。

```typescript
 * 與 Q&A 使用同一個 Dify App，以 mode 在 Dify 端分流至不同 LLM 節點。
 *
 * ⚠️ 前置條件：Dify App 的 Start 節點需包含 mode、transcript、meeting_name 輸入變數，
 * 並設有 If-Else 條件節點，將 mode="summary" 路由至摘要 LLM 節點。
 * 摘要 LLM 節點回傳 JSON：{"summary": "...", "action_items": [...]}
 */
export async function generateSummary(params: {
  transcript: string      // 全量逐字稿（格式："[speaker]: text\n..."）
  meetingName: string
  meetingInstanceId: string   // 作為 Dify user 識別符（用於 Dify 用量追蹤）
}): Promise<{ summary: string; actionItems: string[] }> {
  const res = await fetch(`${DIFY_BASE}/chat-messages`, {
    method: 'POST',
    headers: {
      Authorization: `Bearer ${WORKFLOW_KEY}`,
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      inputs: {
        mode: 'summary',
        transcript: params.transcript,
        meeting_name: params.meetingName,
      },
      query: '請生成會議摘要',   // Chatflow 需要 query 欄位；此為 trigger，實際由 inputs 驅動
      response_mode: 'blocking',
      user: params.meetingInstanceId,
    }),
    signal: AbortSignal.timeout(60_000),   // 摘要較長，給 60 秒
  })
  if (!res.ok) {
    throw new Error(`Dify summary error: ${res.status} ${await res.text()}`)
  }
  const data = await res.json()
  const text: string = data.answer ?? ''
  try {
    const jsonMatch = text.match(/\{[\s\S]*\}/)
    if (!jsonMatch) throw new Error('no JSON in response')
    const parsed = JSON.parse(jsonMatch[0])
    return { summary: parsed.summary ?? '', actionItems: parsed.action_items ?? [] }
  } catch {
    return { summary: text, actionItems: [] }   // fallback：整段文字作為 summary
  }
}
```

```typescript
import { prisma } from '../lib/prisma'
```

## 八、服務重啟恢復策略

服務重啟後，記憶體中的 `activeSessions` 會消失。啟動時依序執行兩個階段：

```typescript
// session-manager.ts
export async function restoreActiveSessions(): Promise<void> {

  // ── 階段一：清理 zombie PENDING ───────────────────────────────────
  //
  // PENDING 是建立流程的暫態，正常情況下數秒內轉為 ACTIVE 或留在 PENDING 等重試。
  // 若重啟後仍為 PENDING 且超過閾值，代表步驟③~⑤的其中一步在崩潰前未完成：
  //   - Vexa bot 可能在運行（無從確認），但後端 WS session 已遺失
  //   - 前端輪詢到此狀態會永久顯示「Bot 加入中」spinner 且無法恢復
  //
  // 安全處置：標記為 FAILED，讓前端顯示明確的失敗訊息。
  // 閾值設為 5 分鐘：正常建立流程 < 30 秒，5 分鐘後必定是 zombie。
  const PENDING_STALE_MINUTES = 5
  const staleThreshold = new Date(Date.now() - PENDING_STALE_MINUTES * 60 * 1000)

  const staleResult = await prisma.meetingInstance.updateMany({
    where: {
      status: 'PENDING',
      createdAt: { lt: staleThreshold },
    },
    data: { status: 'FAILED', endedAt: new Date() },
  })
  if (staleResult.count > 0) {
    logger.warn(
      { count: staleResult.count },
      'startup: cleaned up zombie PENDING meetings (marked as FAILED)'
    )
  }

  // ── 階段二：恢復 ACTIVE session ──────────────────────────────────
  const activeMeetings = await prisma.meetingInstance.findMany({
    where: { status: 'ACTIVE' },
    include: { project: { select: { difyDatasetId: true } } },
  })

  let restored = 0
  for (const meeting of activeMeetings) {
    // 需要 vexaMeetingId（integer）和 vexaNativeMeetingId（slug）才能重建 session
    if (!meeting.vexaMeetingId || !meeting.vexaNativeMeetingId) {
      logger.warn({ meetingInstanceId: meeting.id }, 'missing vexa IDs, skipping restore')
      continue
    }

    // 透過 creatorApiTokenId（整數 FK ref）查詢實際 token 字串
    // DB 中並無跨 schema FK，但 id 一致，用 raw query 跨 schema 存取 public.api_tokens
    const tokenRows = await prisma.$queryRaw<Array<{ token: string }>>`
      SELECT token FROM public.api_tokens
      WHERE id = ${meeting.creatorApiTokenId}
        AND (expires_at IS NULL OR expires_at > NOW())
      LIMIT 1
    `
    if (!tokenRows.length) {
      logger.warn({ meetingInstanceId: meeting.id }, 'creator token expired, marking meeting as ENDED')
      await prisma.meetingInstance.update({
        where: { id: meeting.id },
        data: { status: 'ENDED', endedAt: new Date() },
      })
      continue
    }

    // 重新建立 WebSocket 訂閱
    // meeting.project 為 null 表示無關聯專案，difyDatasetId 傳 null（Q&A 功能停用）
    await createSession(meeting.id, {
      vexaMeetingId: meeting.vexaMeetingId,
      platform: 'google_meet',               // 目前唯一支援的平台
      nativeMeetingId: meeting.vexaNativeMeetingId,
      difyDatasetId: meeting.project?.difyDatasetId ?? null,
      creatorVexaToken: tokenRows[0].token,
    })
    restored++
  }

  logger.info(
    { staleCleaned: staleResult.count, activeRestored: restored },
    'startup restore completed'
  )
}
```

> **限制**：重啟後記憶體內的 `processedSegmentIds` 與 `isSpeaking` 狀態消失，極短時間內可能出現喚醒詞重複觸發，這是可接受的 trade-off（重啟機率低，且 Dify Chatflow 為無狀態查詢，不受影響）。

## 九、錯誤處理策略

### 9.1 三方操作的 Rollback 原則

| 操作 | 失敗點 | Rollback 行為 |
|------|--------|---------------|
| 上傳檔案 | Storage 上傳後 Dify 失敗 | 刪除 Storage 檔案 |
| 上傳檔案 | Dify 上傳後 DB 寫入失敗 | 刪除 Storage + Dify 文件 |
| 建立專案 | Dify 建立後 DB 寫入失敗 | 刪除 Dify dataset |
| 刪除檔案 | 任一步驟失敗 | 記錄 error log，標記異常，不對使用者拋錯（後台補清理） |

### 9.2 統一錯誤回應（`src/middleware/error-handler.ts`）

```typescript
export const errorHandler = (err: Error, c: Context) => {
  if (err instanceof AppError) {
    return c.json({ error_code: err.code, message: err.message, details: err.details }, err.status)
  }
  logger.error({ err }, 'unhandled error')
  return c.json({ error_code: 'INTERNAL_ERROR', message: '伺服器內部錯誤' }, 500)
}
```
